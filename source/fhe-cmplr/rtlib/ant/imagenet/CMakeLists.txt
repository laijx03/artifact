find_package (OpenCV QUIET)
if (OpenCV_FOUND)
        message (STATUS "Build imagenet require OpenCV")
else ()
        message (WARNING "OpenCV not found. Imagenet won't be built")
        return ()
endif ()

include_directories( ${OpenCV_INCLUDE_DIRS} )

file (GLOB FHERTLIB_IMAGENET_SRC_FILES CONFIGURE_DEPENDS *.cxx)
foreach (app ${FHERTLIB_IMAGENET_SRC_FILES})
	get_filename_component (exe ${app} NAME_WE)
	add_executable (${exe} ${app})
	set_property (TARGET ${exe} PROPERTY RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/dataset)
	target_link_libraries (${exe} ${FHERT_ANT_LIB} ${FHE_LIBS} ${MATH_LIBS} ${OpenCV_LIBS})
endforeach ()
